package ir.mohammadno3ratii.app.dev.digicoinshow.base;

import android.annotation.SuppressLint;
import android.content.Context;
import android.graphics.Canvas;
import android.graphics.Color;
import android.graphics.Paint;
import android.util.AttributeSet;
import android.widget.ImageView;

import androidx.annotation.Nullable;

@SuppressLint("AppCompatCustomView")
public class CanvasView extends ImageView {

    String name = "", tt = "";
    Thread thread;
    Float offset = 0f;
    int  textOffset = 0;
    public CanvasView(Context context) {
        super(context);
    }
    public CanvasView(Context context, @Nullable AttributeSet attrs) {
        super(context, attrs);
    }
    public CanvasView(Context context, @Nullable AttributeSet attrs, int defStyleAttr) {
        super(context, attrs, defStyleAttr);
    }

    public void init(String name , String question){
        this.name = name;
        tt = question;
        runTread();
        postInvalidate();
    }

    private void runTread() {
        thread = new Thread(new Runnable() {
            @Override
            public void run() {
            while (true){
                try {
                    offset++;
                    textOffset = (int) (Math.sin((double) System.currentTimeMillis() / 180.0) * 50);
                    Thread.sleep(20);
                    postInvalidate();
                }catch (Exception e){}
            }
            }
        });
        thread.start();
    }

    @Override
    protected void onDraw(Canvas canvas) {
        super.onDraw(canvas);
        Paint paint = new Paint();
        paint.setColor(Color.parseColor("#ff0000"));
        paint.setTextAlign(Paint.Align.CENTER);
        paint.setTextSize(30);
        paint.setStyle(Paint.Style.FILL);
        paint.setAntiAlias(true);

        Paint paint1 = new Paint();
        paint.setColor(Color.parseColor("#ffff00"));
        paint.setStyle(Paint.Style.FILL);
        paint.setAntiAlias(true);

        canvas.drawCircle((1 * (getWidth() / 4)), (200 + offset), 200, paint1);
        canvas.drawCircle((3*(getWidth()/4)), (200 + offset),200,paint1);

        canvas.drawText(name,(1*(getWidth()/4)),(200 + offset),paint);
        canvas.drawText(G.test,(3*(getWidth()/4)), (200 + offset),paint);

        canvas.drawText(tt,(getWidth()/2 ), (getHeight()/2) + textOffset,paint);

    }
}
